    <Steps>
      <Step title="Install the package">
        Add Mantle Polaris to your Shopify app project:

        ```bash npm
        npm install @heymantle/polaris
        ```

        The package requires React 18+ and Shopify Polaris 13+.
      </Step>

      <Step title="Wrap your app with MantleProvider">
        Add the Mantle context provider to your app:

        ```jsx
        import { MantleProvider } from "@heymantle/react";

        function App() {
          return (
            <MantleProvider
              appId={process.env.MANTLE_APP_ID}
              customerApiToken={shop.mantleApiToken}
            >
              <YourApp />
            </MantleProvider>
          );
        }
        ```
      </Step>

      <Step title="Add your first component">
        Display plan options to your users:

        ```jsx
        import { HorizontalPlanCards } from "@heymantle/polaris";
        import { useMantle } from "@heymantle/react";

        function PricingPage() {
          const { customer, plans, subscribe } = useMantle();

          return (
            <HorizontalPlanCards
              customer={customer}
              plans={plans}
              onSubscribe={async ({ planId, discountId }) => {
                const subscription = await subscribe({ planId, discountId });
                window.open(subscription.confirmationUrl, "_top");
              }}
            />
          );
        }
        ```

        <Note>
          The component automatically handles monthly/yearly toggles, trial periods, discounts, and plan recommendations.
        </Note>
      </Step>
    </Steps>
